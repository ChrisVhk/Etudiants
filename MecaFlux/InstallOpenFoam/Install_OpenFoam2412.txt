# Ajouter la clé GPG
wget -q -O - https://dl.openfoam.com/add-debian-repo.sh | sudo bash

# Ou manuellement :
curl https://dl.openfoam.com/gpg.key | sudo apt-key add -
sudo add-apt-repository "http://dl.openfoam.com/ubuntu $(lsb_release -cs) main"

sudo apt update
sudo apt upgrade

# Rechercher tous les paquets OpenFOAM
apt search openfoam | grep -E "openfoam[0-9]+"

# Ou plus spécifiquement pour les versions ESI
apt search openfoam2 | sort

# Pour voir les informations sur une version spécifique
apt show openfoam2412-dev
apt show openfoam2412-default


# Installer OpenFOAM v2412 version par défaut
sudo apt install openfoam2412-default

# Installer ParaView (optionnel mais recommandé)
sudo apt install paraviewopenfoam2412-default

# Ajouter au fichier .bashrc
echo "source /usr/lib/openfoam/openfoam2412/etc/bashrc" >> ~/.bashrc

# Recharger le bashrc
source ~/.bashrc

# Vérifier la version installée
foamVersion
